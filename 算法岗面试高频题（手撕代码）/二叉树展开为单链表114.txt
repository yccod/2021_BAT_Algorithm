class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        if not root:
            return root
        stack=[root]
        pre=None
        while stack:
            node=stack.pop()
            if pre:
                pre.left=None
                pre.right=node
            if node.right:
                stack.append(node.right)
            if node.left:
                stack.append(node.left)
            pre=node
        return root
